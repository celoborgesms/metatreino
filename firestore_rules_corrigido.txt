rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function userEmail() {
      return signedIn() && request.auth.token.email != null
        ? request.auth.token.email
        : "";
    }

    function isOwner(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    // Regra fixa do app:
    // - documentos de /admins usam o e-mail do treinador como ID, em minúsculas.
    // - o código do app consulta /admins/{emailDoTreinador}.
    // - crie esse documento manualmente pelo console do Firebase com ativo: true.
    function isAdmin() {
      return signedIn()
        && request.auth.token.email != null
        && exists(/databases/$(database)/documents/admins/$(userEmail()))
        && get(/databases/$(database)/documents/admins/$(userEmail())).data.ativo == true;
    }

    match /usuarios/{userId} {
      allow read, create, update, delete: if isOwner(userId) || isAdmin();

      match /{subdocumentos=**} {
        allow read, create, update, delete: if isOwner(userId) || isAdmin();
      }
    }

    // O app consulta usuariosAutorizados pelo e-mail em minúsculas.
    // Use o e-mail do aluno como ID do documento.
    //
    // CORREÇÃO em relação ao arquivo original: a regra antiga só deixava
    // cada aluno ler o PRÓPRIO documento (para checar seu acesso), mas não
    // dava ao admin permissão de LISTAR a coleção inteira — e é isso que o
    // painel do treinador precisa fazer para mostrar a lista de alunos.
    // Por isso adicionamos "isAdmin() ||" na regra de leitura abaixo.
    match /usuariosAutorizados/{emailDoc} {
      allow read: if isAdmin() || (
        signedIn()
        && request.auth.token.email != null
        && request.auth.token.email_verified == true
        && userEmail() == emailDoc
      );
      allow create, update, delete: if isAdmin();
    }

    // Documento administrativo criado fora do app, pelo console do Firebase.
    // ID recomendado: e-mail do treinador em minúsculas.
    match /admins/{emailDoc} {
      allow read: if signedIn()
        && request.auth.token.email != null
        && userEmail() == emailDoc;
      allow create, update, delete: if false;
    }

    match /{documento=**} {
      allow read, write: if false;
    }
  }
}
